To build responsive UI's, you first need to define your breakpoints in the `screens` object:

```jsx
// interop.ts
import { createStyled } from '@interop/react';

const { styled, css } = createStyled({
  screens: {
    small: (rule) => rule,
    medium: (rule) => `@media (min-width: 480px) { ${rule} }`,
    large: (rule) => `@media (min-width: 960px) { ${rule} }`,
  },
});

export { styled, css };
```

Once created, you can use them in the Style Objects:

```jsx
import { styled } from './interop.ts';

const Layout = styled('div', {
  display: 'flex'
  small: {
    flexDirection: 'column'
  },
  medium: {
    flexDirection: 'row'
  }
});
```

You can also use screens when applying variants:

```jsx
// interop.ts
import { styled } from './interop.ts'

const Button = styled('button', {
  appearance: 'none',
  color: 'white'
}, {
  variant: {
    primary: {
      backgroundColor: 'red',
    },
    secondary: {
      backgroundColor: 'blue',
    },
  }
})

() => <>
  <Button variant={small: 'primary', medium: 'secondary'}>Responsive button</Button>
</>
```
